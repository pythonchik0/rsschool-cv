def caps(n):
    for i in range(len(n)):
        if i == 0:
            [i] = chr(ord(n[i])-32)
        elif n[i] == ' ':
            n[i+1] = chr(ord(n[i+1])-32)
            return n
m = list(input())
print(''.join(caps(m)))
